CREATE OR ALTER PROCEDURE srv.ExcelGetDBFilesOperationsStat
as
begin

/* возвращает информацию по БД */
SET NOCOUNT ON;
SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED; 

SELECT @@SERVERNAME as [ServerName]
,[DBName]
      ,[FileId]
      ,[NumberReads]
      ,[BytesRead]
      ,[IoStallReadMS]
      ,[NumberWrites]
      ,[BytesWritten]
      ,[IoStallWriteMS]
      ,[IoStallMS]
      ,[BytesOnDisk]
      ,[Type_desc]
      ,[FileName]
      ,[Drive]
      ,[Physical_Name]
      ,[Ext]
      ,[CountPage]
      ,[SizeMb]
      ,[SizeGb]
      ,[Growth]
      ,[GrowthMb]
      ,[GrowthGb]
      ,[GrowthPercent]
      ,[is_percent_growth]
      ,[StateDesc]
      ,[IsMediaReadOnly]
      ,[IsReadOnly]
      ,[IsSpace]
      ,[IsNameReserved]
  FROM [inf].[vDBFilesOperationsStat];
end
GO
EXEC sys.sp_addextendedproperty N'MS_Description', N'Возвращает информацию по БД', 'SCHEMA', N'srv', 'PROCEDURE', N'ExcelGetDBFilesOperationsStat'
GO
