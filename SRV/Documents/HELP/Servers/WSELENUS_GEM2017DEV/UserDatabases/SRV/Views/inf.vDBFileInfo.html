SQL скрипт

SET QUOTED_IDENTIFIER, ANSI_NULLS ON
GO
create OR ALTER view inf.vDBFileInfo as
--информация о файлах БД
SELECT  @@Servername AS Server ,
        DB_NAME() AS DB_Name ,
        File_id ,
        Type_desc ,
        Name ,
        LEFT(Physical_Name, 1) AS Drive ,
        Physical_Name ,
        RIGHT(physical_name, 3) AS Ext ,
        Size ,
        Growth
FROM    sys.database_files
--ORDER BY File_id;
GO
EXEC sys.sp_addextendedproperty N'MS_Description', N'Информация о файлах БД', 'SCHEMA', N'inf', 'VIEW', N'vDBFileInfo'
GO
