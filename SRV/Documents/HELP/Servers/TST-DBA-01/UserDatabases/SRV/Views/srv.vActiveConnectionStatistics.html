SQL Script

SET QUOTED_IDENTIFIER, ANSI_NULLS ON
GO
CREATE OR ALTER view srv.vActiveConnectionStatistics as
SELECT [ServerName]
,[DBName]
      ,[ProgramName]
      ,[LoginName]
      ,[Status]
      ,[SessionID]
,[LoginTime]
,[EndRegUTCDate]
,[InsertUTCDate]
  FROM [srv].[ActiveConnectionStatistics] with(readuncommitted)
GO
EXEC sys.sp_addextendedproperty N'MS_Description', N'Информация по активным подключениям по собранным данных', 'SCHEMA', N'srv', 'VIEW', N'vActiveConnectionStatistics'
GO
