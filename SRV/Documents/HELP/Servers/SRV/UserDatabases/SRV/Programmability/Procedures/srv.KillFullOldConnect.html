CREATE OR ALTER PROCEDURE srv.KillFullOldConnect
 @OldHour int=24
AS
BEGIN
 
/* Удаляет те подключения, последнее выполнение которых было более суток назад. Внимание! Системные БД master, tempdb, model и msdb не участвуют в процессе. Однако, БД distribution для репликаций будет затронута и это нормально. */
 SET NOCOUNT ON;
 SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED;

 declare @query nvarchar(max);
 set @query = '';

 select @query=coalesce(@query,',' )
    +'kill '
    +convert(varchar, spid)
    +'; '
 from master..sysprocesses
 where dbid>4
 and [last_batch]<dateadd(hour,-@OldHour,getdate())
 order by [last_batch]
 
 if len(@query) > 0
 begin
  begin try
   exec(@query);
  end try
  begin catch
  end catch
 end
END
GO

EXEC sys.sp_addextendedproperty N'MS_Description', N'Удаляет те подключения, последнее выполнение которых было более суток назад', 'SCHEMA', N'srv', 'PROCEDURE', N'KillFullOldConnect'
GO
