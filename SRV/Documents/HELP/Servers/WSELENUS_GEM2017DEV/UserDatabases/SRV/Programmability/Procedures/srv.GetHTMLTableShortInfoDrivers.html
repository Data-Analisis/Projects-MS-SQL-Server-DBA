CREATE OR ALTER PROCEDURE srv.GetHTMLTableShortInfoDrivers
@body nvarchar(max) OUTPUT
AS
BEGIN

/* формирует HTML-код для таблицы запоминающих устройств (жестких дисков) */
SET NOCOUNT ON;
SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED;
declare @tbl table (
Driver_GUID uniqueidentifier
,[Name] nvarchar(255)
,[TotalSpaceGb] float
,[FreeSpaceGb] float
,[DiffFreeSpaceMb] float
,[FreeSpacePercent] float
,[DiffFreeSpacePercent] float
,UpdateUTCDate datetime
,[Server] nvarchar(255)
,ID int identity(1,1)
);
declare
@Driver_GUID uniqueidentifier
,@Name nvarchar(255)
,@TotalSpaceGb float
,@FreeSpaceGb float
,@DiffFreeSpaceMb float
,@FreeSpacePercent float
,@DiffFreeSpacePercent float
,@UpdateUTCDate datetime
,@Server nvarchar(255)
,@ID int;
insert into @tbl(
Driver_GUID
,[Name]
,[TotalSpaceGb]
,[FreeSpaceGb]
,[DiffFreeSpaceMb]
,[FreeSpacePercent]
,[DiffFreeSpacePercent]
,UpdateUTCDate
,[Server]
)
select Driver_GUID
,[Name]
,[TotalSpaceGb]
,[FreeSpaceGb]
,[DiffFreeSpaceMb]
,[FreeSpacePercent]
,[DiffFreeSpacePercent]
,UpdateUTCDate
,[Server]
from srv.vDrivers
where [DiffFreeSpacePercent]<=-5
or [FreeSpacePercent]<=15
order by [Server] asc, [Name] asc;
if(exists(select top(1) 1 from @tbl))
begin
set @body='В ходе анализа были выявлены следующие носители иформации, у которых либо свободного объема осталось меньше 15%, либо свободное место уменьшается свыше 5% за день:

'+'';
set @body=@body+'';
set @body=@body+'';

set @body=@body+'';
set @body=@body+'';
set @body=@body+'';

set @body=@body+'';
set @body=@body+'';
set @body=@body+'';
set @body=@body+'';
set @body=@body+'';
set @body=@body+'';
set @body=@body+'';
while((select top 1 1 from @tbl)>0)
begin
set @body=@body+'';
select top 1
@Driver_GUID = Driver_GUID
,@Name = Name
,@TotalSpaceGb = TotalSpaceGb
,@FreeSpaceGb = FreeSpaceGb
,@DiffFreeSpaceMb = DiffFreeSpaceMb
,@FreeSpacePercent = FreeSpacePercent
,@DiffFreeSpacePercent = DiffFreeSpacePercent
,@UpdateUTCDate = UpdateUTCDate
,@Server = [Server]
,@ID = [ID]
from @tbl;
set @body=@body+'';

set @body=@body+'';

set @body=@body+'';

set @body=@body+'';
set @body=@body+'';
set @body=@body+'';
set @body=@body+'';
set @body=@body+'';
set @body=@body+'';
set @body=@body+'';
delete from @tbl
where ID=@ID;
set @body=@body+'';
end
set @body=@body+'
';
set @body=@body+'№ п/п';
set @body=@body+'		';
set @body=@body+'ГУИД';
set @body=@body+'		';
set @body=@body+'СЕРВЕР';
set @body=@body+'		';
set @body=@body+'ТОМ';
set @body=@body+'		';
set @body=@body+'ЕМКОСТЬ, ГБ.';
set @body=@body+'		';
set @body=@body+'СВОБОДНО, ГБ.';
set @body=@body+'		';
set @body=@body+'ИЗМЕНЕНИЕ СВОБОДНОГО МЕСТА, МБ.';
set @body=@body+'		';
set @body=@body+'СВОБОДНО, %';
set @body=@body+'		';
set @body=@body+'ИЗМЕНЕНИЕ СВОБОДНОГО МЕСТА, %';
set @body=@body+'		';
set @body=@body+'UTC ВРЕМЯ ОБНАРУЖЕНИЯ';
set @body=@body+'
';
set @body=@body+cast(@ID as nvarchar(max));
set @body=@body+'		';
set @body=@body+cast(@Driver_GUID as nvarchar(255));
set @body=@body+'		';
set @body=@body+coalesce(@Server,'');
set @body=@body+'		';

set @body=@body+cast(@TotalSpaceGb as nvarchar(255));

set @body=@body+'		';
set @body=@body+cast(@DiffFreeSpaceMb as nvarchar(255));
set @body=@body+'		';
set @body=@body+cast(@FreeSpacePercent as nvarchar(255));
set @body=@body+'		';
set @body=@body+cast(@DiffFreeSpacePercent as nvarchar(255));
set @body=@body+'		';
set @body=@body+rep.GetDateFormat(@UpdateUTCDate, default)+' '+rep.GetTimeFormat(@UpdateUTCDate, default);
set @body=@body+'
';
set @body=@body+'

Для более детальной информации обратитесь к представлению SRV.srv.vDrivers

end

--end
END
GO
