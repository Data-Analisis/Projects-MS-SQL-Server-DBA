GO




CREATE OR ALTER view inf.vScheduleMultiJobs as
/* ГЕМ: расписания, у которых несколько задач */
with sh as(
  SELECT schedule_id
  FROM [inf].[vJobSchedules]
  group by schedule_id
  having count(*)>1
)
select *
from msdb.dbo.sysschedules as s
where exists(select top(1) 1 from sh where sh.schedule_id=s.schedule_id)



GO

EXEC sys.sp_addextendedproperty N'MS_Description', N'Расписания, у которых несколько задач экземпляра MS SQL Server', 'SCHEMA', N'inf', 'VIEW', N'vScheduleMultiJobs'
GO
