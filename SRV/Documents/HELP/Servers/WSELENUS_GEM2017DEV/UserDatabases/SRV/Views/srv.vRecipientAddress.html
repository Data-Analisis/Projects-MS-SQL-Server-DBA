SQL скрипт

SET QUOTED_IDENTIFIER, ANSI_NULLS ON
GO
create OR ALTER view srv.vRecipientAddress as
select t1.Address, 
t2.Recipient_Code, 
t2.Recipient_Name, 
t1.IsDeleted as AddressIsDeleted,
t2.IsDeleted as RecipientIsDeleted
from [srv].[Address] as t1
inner join [srv].[Recipient] as t2 on t1.Recipient_GUID=t2.Recipient_GUID
GO
EXEC sys.sp_addextendedproperty N'MS_Description', N'Адресаты с почтовыми адресами', 'SCHEMA', N'srv', 'VIEW', N'vRecipientAddress'
GO
