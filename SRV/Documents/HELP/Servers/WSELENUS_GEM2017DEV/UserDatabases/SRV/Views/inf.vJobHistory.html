SQL скрипт

SET QUOTED_IDENTIFIER, ANSI_NULLS ON
GO
create OR ALTER view inf.vJobHistory as
SELECT [instance_id]
      ,[job_id]
      ,[step_id]
      ,[step_name]
      ,[sql_message_id]
      ,[sql_severity]
      ,[message]
      ,[run_status]
      ,[run_date]
      ,[run_time]
      ,[run_duration]
      ,[operator_id_emailed]
      ,[operator_id_netsent]
      ,[operator_id_paged]
      ,[retries_attempted]
      ,[server]
  FROM [msdb].[dbo].[sysjobhistory];
GO
EXEC sys.sp_addextendedproperty N'MS_Description', N'История выполнения заданий Агента экземпляра MS SQL Server', 'SCHEMA', N'inf', 'VIEW', N'vJobHistory'
GO
