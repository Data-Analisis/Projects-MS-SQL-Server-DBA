SET QUOTED_IDENTIFIER, ANSI_NULLS ON
GO
create OR ALTER view inf.vJobSchedules as
SELECT [schedule_id]
      ,[job_id]
      ,[next_run_date]
      ,[next_run_time]
  FROM [msdb].[dbo].[sysjobschedules];
GO
EXEC sys.sp_addextendedproperty N'MS_Description', N'Расписания заданий Агента экземпляра MS SQL Server', 'SCHEMA', N'inf', 'VIEW', N'vJobSchedules'
GO
