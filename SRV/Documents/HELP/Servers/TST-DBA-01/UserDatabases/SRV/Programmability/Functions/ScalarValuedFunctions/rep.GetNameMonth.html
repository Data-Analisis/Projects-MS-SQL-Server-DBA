CREATE OR ALTER FUNCTION rep.GetNameMonth
(
@Month_Num int -- заданный номер месяца
)
RETURNS nvarchar(256)
AS
/* 15.04.2014 ГЕМ: Возвращает русское название месяца по заданному его номеру */
BEGIN
declare @Month_Name nvarchar(256);
set @Month_Name=case @Month_Num
when 1 then 'январь'
when 2 then 'февраль'
when 3 then 'март'
when 4 then 'апрель'
when 5 then 'май'
when 6 then 'июнь'
when 7 then 'июль'
when 8 then 'август'
when 9 then 'сентябрь'
when 10 then 'октябрь'
when 11 then 'ноябрь'
when 12 then 'декабрь'
end
return @Month_Name;
END
GO
EXEC sys.sp_addextendedproperty N'MS_Description', N'Возвращает русское название месяца по заданному его номеру', 'SCHEMA', N'rep', 'FUNCTION', N'GetNameMonth'
GO
EXEC sys.sp_addextendedproperty N'MS_Description', N'Номер месяца в году', 'SCHEMA', N'rep', 'FUNCTION', N'GetNameMonth', 'PARAMETER', N'@Month_Num'
GO
