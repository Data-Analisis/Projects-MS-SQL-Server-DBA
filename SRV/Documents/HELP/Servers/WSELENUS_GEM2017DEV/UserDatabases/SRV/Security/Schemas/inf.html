Описание

Объекты схемы inf базируются на встроенных объектах и данную информацию можно получить в любой момент времени (т е напрямую запросом без использования объекта в этой схеме)

SQL скрипт

CREATE SCHEMA inf AUTHORIZATION dbo
GO
EXEC sys.sp_addextendedproperty N'MS_Description', N'Объекты схемы inf базируются на встроенных объектах и данную информацию можно получить в любой момент времени (т е напрямую запросом без использования объекта в этой схеме)', 'SCHEMA', N'inf'
GO
