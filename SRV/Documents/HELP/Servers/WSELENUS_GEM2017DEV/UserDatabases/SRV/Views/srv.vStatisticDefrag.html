SQL скрипт

SET QUOTED_IDENTIFIER, ANSI_NULLS ON
GO
CREATE OR ALTER view srv.vStatisticDefrag as
SELECT top 1000
[db]
,[shema]
      ,[table]
      ,[IndexName]
      ,avg([frag]) as AvgFrag
      ,avg([frag_after]) as AvgFragAfter
,avg(page) as AvgPage
  FROM [srv].[Defrag]
  group by [db], [shema], [table], [IndexName]
  order by abs(avg([frag])-avg([frag_after])) desc
GO
EXEC sys.sp_addextendedproperty N'MS_Description', N'Информация о выполнении реорганизации индексов по собранным данным', 'SCHEMA', N'srv', 'VIEW', N'vStatisticDefrag'
GO
