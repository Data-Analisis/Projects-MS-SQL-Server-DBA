CREATE OR ALTER FUNCTION rep.GetNumMonth
(
 @Month_Name nvarchar(256) -- заданное название месяца по-русски
)
RETURNS int
AS
/* 15.04.2014 ГЕМ: Возвращает номер месяца по его заданному названию по-русски */
BEGIN
 declare @Month_Num int;
 set @Month_Num=case ltrim(rtrim(@Month_Name))
     when 'январь' then 1
     when 'февраль' then 2
     when 'март' then 3
     when 'апрель' then 4
     when 'май' then 5
     when 'июнь' then 6
     when 'июль' then 7
     when 'август' then 8
     when 'сентябрь' then 9
     when 'октябрь' then 10
     when 'ноябрь' then 11
     when 'декабрь' then 12
     end
 return @Month_Num;

END


GO

EXEC sys.sp_addextendedproperty N'MS_Description', N'Возвращает номер месяца по его заданному названию по-русски', 'SCHEMA', N'rep', 'FUNCTION', N'GetNumMonth'
GO

EXEC sys.sp_addextendedproperty N'MS_Description', N'Название месяца', 'SCHEMA', N'rep', 'FUNCTION', N'GetNumMonth', 'PARAMETER', N'@Month_Name'
GO
