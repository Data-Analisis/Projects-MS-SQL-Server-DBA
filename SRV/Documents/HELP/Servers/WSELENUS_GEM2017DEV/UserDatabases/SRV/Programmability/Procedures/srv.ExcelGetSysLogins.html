Описание

Возвращает все логины для входа на сервер MS SQL SERVER

SQL скрипт

SET QUOTED_IDENTIFIER, ANSI_NULLS ON
GO
CREATE OR ALTER PROCEDURE srv.ExcelGetSysLogins
as
begin

/* возвращает все логины для входа на сервер MS SQL SERVER */
SET NOCOUNT ON;
SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED; 

SELECT @@SERVERNAME as [ServerName]
,[Name] as [Login]
      ,[CreateDate]
      ,[UpdateDate]
      ,[DBName]
      ,[DenyLogin]
      ,[HasAccess]
,case when([IsntName]=1) then N'MS SQL SERVER' else N'WINDOWS' end as [NameInput]
      ,[IsntGroup] as [GroupWindows]
      ,[IsntUser] as [UserWindows]
      ,[SysAdmin]
      ,[SecurityAdmin]
      ,[ServerAdmin]
      ,[SetupAdmin]
      ,[ProcessAdmin]
      ,[DiskAdmin]
      ,[DBCreator]
      ,[BulkAdmin]
FROM [inf].[SysLogins];
end
GO
EXEC sys.sp_addextendedproperty N'MS_Description', N'Возвращает все логины для входа на сервер MS SQL SERVER', 'SCHEMA', N'srv', 'PROCEDURE', N'ExcelGetSysLogins'
GO
