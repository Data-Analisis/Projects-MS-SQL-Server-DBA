SQL скрипт

SET QUOTED_IDENTIFIER, ANSI_NULLS ON
GO
create OR ALTER view srv.vDBFiles as
SELECT [DBFile_GUID]
      ,[Server]
      ,[Name]
      ,[Drive]
      ,[Physical_Name]
      ,[Ext]
      ,[Growth]
      ,[IsPercentGrowth]
      ,[DB_ID]
,[File_ID]
      ,[DB_Name]
      ,[SizeMb]
      ,[DiffSizeMb]
,round([SizeMb]/1024,3) as [SizeGb]
      ,round([DiffSizeMb]/1024,3) as [DiffSizeGb]
,round([SizeMb]/1024/1024,3) as [SizeTb]
      ,round([DiffSizeMb]/1024/1024,3) as [DiffSizeTb]
,round([DiffSizeMb]/([SizeMb]/100), 3) as [DiffSizePercent]
      ,[InsertUTCDate]
      ,[UpdateUTCdate]
  FROM [srv].[DBFile];
GO
EXEC sys.sp_addextendedproperty N'MS_Description', N'Информация по файлам всех БД экземпляра MS SQL Server', 'SCHEMA', N'srv', 'VIEW', N'vDBFiles'
GO
