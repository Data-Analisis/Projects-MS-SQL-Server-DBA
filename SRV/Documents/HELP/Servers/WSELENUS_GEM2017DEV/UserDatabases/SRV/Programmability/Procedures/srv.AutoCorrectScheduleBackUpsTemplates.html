CREATE OR ALTER PROCEDURE srv.AutoCorrectScheduleBackUpsTemplates
@path_base nvarchar(4000)='\\backup-srv01\SQL_Backups\DB\PRD_0'--'dir \\backup-srv01\SQL_Backups\DB\PRD_0' --путь
,@dbname nvarchar(255) --БД
,@job_id uniqueidentifier --задание, расписание которого необходимо скорректировать
AS
BEGIN
SET NOCOUNT ON;
SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED;
if((@path_base is null) or (@dbname is null) or (@job_id is null)) THROW 51000, 'Не все параметры заданы!', 1; 
--declare @temptable table ([text] nvarchar(200));

--declare @path nvarchar(4000);
--declare @dt0 datetime;
declare @schedule_id int;
declare @job_name nvarchar(255);
declare @diffsec int;
declare @active_start_time int;
declare @BackupFinishDate datetime;
declare @hh_active_start_time int;
declare @mm_active_start_time int;
declare @str_hh_active_start_time nvarchar(255);
declare @str_mm_active_start_time nvarchar(255);
declare @str_ss_active_start_time nvarchar(255);
declare @str_active_start_time nvarchar(255);

----собираем самый новый по модификации файл резервной копии всех БД с каждого каталога
--while(@ind_max>0)

-- set @ind_max=@ind_max-1;

----берем самую свежую дату из таких файлов
--set @dt0=(select max([DT]) from @tbl_dt);

--по ИД задания Агента получаем ИД расписания
set @schedule_id=(
select top(1) [schedule_id]
from [msdb]..[sysjobschedules]
where [job_id]=@job_id
);

--получаем время в секундах, потраченное на создание последней полной резервной копии указанной БД
set @diffsec=(
select top(1) DateDiff(second, [BackupStartDate], [BackupFinishDate])
from [inf].[vServerBackupDB]
where BackupType='database'
and [DBName]=@dbname
and [PhysicalDeviceName] like @path_base+'%'
order by [BackupStartDate] desc
);

----получаем время старта задания Агента
--set @active_start_time=(
-- select [active_start_time]
-- from [msdb]..[sysschedules]
-- where [schedule_id]=@schedule_id
--);

--вычисляем время старта задания Агента
--берем всегда, что к 05:00:00 все резервные копии должны быть готовы
set @active_start_time=5*3600-@diffsec;
--время запуска не ранее 02:46:02
if(@active_start_time<10000) set @active_start_time=10000;
set @hh_active_start_time=@active_start_time/3600;
set @mm_active_start_time=@active_start_time/60;

set @str_hh_active_start_time=cast(@hh_active_start_time as nvarchar(255));
set @str_mm_active_start_time=cast(@mm_active_start_time-@hh_active_start_time*60 as nvarchar(255));
set @str_ss_active_start_time=cast(@active_start_time-@mm_active_start_time*60 as nvarchar(255));

if((len(@str_mm_active_start_time)<2) and (@hh_active_start_time>0)) set @str_mm_active_start_time='0'+@str_mm_active_start_time;
if((len(@str_ss_active_start_time)<2) and (@mm_active_start_time>0)) set @str_ss_active_start_time='0'+@str_ss_active_start_time;

set @str_active_start_time=@str_hh_active_start_time+@str_mm_active_start_time+@str_ss_active_start_time;

set @active_start_time=cast(@str_active_start_time as int);

--непосредственно корректируем время старта задания Агента
EXEC [msdb]..sp_attach_schedule @job_id=@job_id,@schedule_id=@schedule_id;
EXEC [msdb]..sp_update_schedule @schedule_id=@schedule_id, @active_start_time=@active_start_time;
END
