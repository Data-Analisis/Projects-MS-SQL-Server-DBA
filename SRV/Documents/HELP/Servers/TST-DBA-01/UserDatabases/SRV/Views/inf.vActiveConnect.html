SQL Script

SET QUOTED_IDENTIFIER, ANSI_NULLS ON
GO
CREATE OR ALTER view inf.vActiveConnect as
select
    db_name([database_id]) as DBName,
    [session_id] as SessionID,
    [login_name] as LoginName,
[program_name] as ProgramName,
[status] as [Status],
[login_time] as [LoginTime]
from
    sys.dm_exec_sessions
where
    [database_id] > 0
GO
EXEC sys.sp_addextendedproperty N'MS_Description', N'Активные сессии экземпляра MS SQL Server (sys.dm_exec_sessions)', 'SCHEMA', N'inf', 'VIEW', N'vActiveConnect'
GO
