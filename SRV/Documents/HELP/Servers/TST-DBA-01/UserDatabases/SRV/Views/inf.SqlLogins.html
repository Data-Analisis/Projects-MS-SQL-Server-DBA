SET QUOTED_IDENTIFIER, ANSI_NULLS ON
GO
create OR ALTER view inf.SqlLogins
as
/* логи скульные на вход */
select [name] as [Name], 
create_date as CreateDate, 
modify_date as ModifyDate, 
default_database_name as DefaultDatabaseName,

GO
