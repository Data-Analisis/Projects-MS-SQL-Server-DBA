SQL скрипт

SET QUOTED_IDENTIFIER, ANSI_NULLS ON
GO
create OR ALTER view inf.SqlLogins
as
/* логи скульные на вход */
select [name] as [Name], 
create_date as CreateDate, 
modify_date as ModifyDate, 
default_database_name as DefaultDatabaseName, 
is_policy_checked as IsPolicyChecked, --политика паролей
is_expiration_checked as IsExpirationChecked, --истечение срока действия паролей
is_disabled as IsDisabled
from sys.sql_logins
GO
EXEC sys.sp_addextendedproperty N'MS_Description', N'Логи скульные на вход экземпляра MS SQL Server (sys.sql_logins)', 'SCHEMA', N'inf', 'VIEW', N'SqlLogins'
GO
