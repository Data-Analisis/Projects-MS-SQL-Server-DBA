SQL скрипт

CREATE TABLE srv.TableIndexStatistics (
  ServerName nvarchar(255) NOT NULL,
  DBName nvarchar(255) NULL,
  SchemaName nvarchar(255) NULL,
  TableName nvarchar(255) NULL,
  IndexUsedForCounts nvarchar(255) NULL,
  CountRows bigint NULL,
  InsertUTCDate datetime NOT NULL CONSTRAINT DF_TableIndexStatistics_InsertUTCDate DEFAULT (getutcdate())
)
ON [PRIMARY]
GO
CREATE CLUSTERED INDEX indInsertUTCDate
  ON srv.TableIndexStatistics (InsertUTCDate)
  WITH (FILLFACTOR = 95)
  ON [PRIMARY]
GO
EXEC sys.sp_addextendedproperty N'MS_Description', N'Информация о размерах индексов таблицы на каждый момент времени экземпляра MS SQL Server', 'SCHEMA', N'srv', 'TABLE', N'TableIndexStatistics'
GO
EXEC sys.sp_addextendedproperty N'MS_Description', N'Сервер', 'SCHEMA', N'srv', 'TABLE', N'TableIndexStatistics', 'COLUMN', N'ServerName'
GO
EXEC sys.sp_addextendedproperty N'MS_Description', N'БД', 'SCHEMA', N'srv', 'TABLE', N'TableIndexStatistics', 'COLUMN', N'DBName'
GO
EXEC sys.sp_addextendedproperty N'MS_Description', N'Схема', 'SCHEMA', N'srv', 'TABLE', N'TableIndexStatistics', 'COLUMN', N'SchemaName'
GO
EXEC sys.sp_addextendedproperty N'MS_Description', N'Таблица', 'SCHEMA', N'srv', 'TABLE', N'TableIndexStatistics', 'COLUMN', N'TableName'
GO
