CREATE OR ALTER PROCEDURE srv.ExcelGetWaits
as
begin
 
/* возвращает задержки по статистическим данным MS SQL Server */
 SET NOCOUNT ON;
 SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED; 
 
 SELECT @@SERVERNAME as [ServerName]
   ,[WaitType]
      ,[Wait_S]
      ,[Resource_S]
      ,[Signal_S]
      ,[WaitCount]
      ,[Percentage]
      ,[AvgWait_S]
      ,[AvgRes_S]
      ,[AvgSig_S]
 FROM [inf].[vWaits];
end
GO

EXEC sys.sp_addextendedproperty N'MS_Description', N'Возвращает задержки по статистическим данным MS SQL Server', 'SCHEMA', N'srv', 'PROCEDURE', N'ExcelGetWaits'
GO
