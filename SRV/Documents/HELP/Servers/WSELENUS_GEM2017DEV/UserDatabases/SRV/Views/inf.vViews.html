SQL скрипт

SET QUOTED_IDENTIFIER, ANSI_NULLS ON
GO
create OR ALTER view inf.vViews as
--представления
select @@Servername AS Server,
       DB_NAME() AS DBName,
s.name as SchemaName,
t.name as TableName,
t.type as Type,
t.type_desc as TypeDesc,
t.create_date as CreateDate,
t.modify_date as ModifyDate
from sys.views as t
inner join sys.schemas as s on t.schema_id=s.schema_id
GO
EXEC sys.sp_addextendedproperty N'MS_Description', N'Представления', 'SCHEMA', N'inf', 'VIEW', N'vViews'
GO
